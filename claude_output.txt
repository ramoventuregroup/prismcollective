function getCheckoutURL(productId, qty = 1) {

/**
 * Render star rating HTML
 * @param {number} rating  – e.g. 4.5
 * @param {number} count   – review count
 */
function renderStars(rating, count) {
  const full = Math.floor(rating);
  const half = rating % 1 >= 0.5;
  const empty = 5 - full - (half ? 1 : 0);
  let stars = '★'.repeat(full);
  if (half) stars += '✦';
  stars += '☆'.repeat(empty);
  const countText = count ? `<span class="review-count">(${count})</span>` : '';
  return `<div class="product-rating">
    <span class="stars" title="${rating} out of 5">${stars}</span>
    ${countText}
  </div>`;
}

/**
 * Format USD price
 * @param {number} price
 */
function formatPrice(price) {
  return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(price);
}

/**
 * Get sale percentage
 */
function getSalePercent(price, compare) {
  if (!compare || compare <= price) return null;
  return Math.round((1 - price / compare) * 100);
}

/**
 * Show toast notification
 * @param {string} message
 * @param {'success'|'error'|'info'} type
 * @param {number} duration  ms

  const toast = document.getElementById('toast');
  if (!toast) return;
  toast.textContent = message;
  toast.className = `toast show ${type}`;
  clearTimeout(toast._timer);
  toast._timer = setTimeout(() => {
    toast.className = 'toast';
  }, duration);
}

/**
 * Debounce utility
 */
function debounce(fn, wait) {
  let t;
  return (...args) => { clearTimeout(t); t = setTimeout(() => fn(...args), wait); };
}

let scrollObserver;

function observeElements(elements) {
  if (!scrollObserver) {
    scrollObserver = new IntersectionObserver(
      (entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            entry.target.classList.add('visible');
            scrollObserver.unobserve(entry.target);
          }
        });
      },
      { threshold: 0.08, rootMargin: '0px 0px -40px 0px' }
    );
  }
  elements.forEach(el => {
    if (el && !el.classList.contains('visible')) {
      scrollObserver.observe(el);
    }
  });
}

function initExitPopup() {
  // Don't show if already shown this session
  if (sessionStorage.getItem(POPUP_KEY)) return;

  const popup = document.getElementById('exitPopup');
  const overlay = document.getElementById('popupOverlay');
  const closeBtn = document.getElementById('popupClose');
  const dismissBtn = document.getElementById('popupDismiss');
  const ctaBtn = document.getElementById('popupCta');

  if (!popup) return;

  let triggered = false;
  let showTimer;

  function showPopup() {
    if (triggered) return;
    triggered = true;
    sessionStorage.setItem(POPUP_KEY, '1');
    popup.classList.add('active');
    if (overlay) overlay.classList.add('active');
    document.body.style.overflow = 'hidden';
  }

  function hidePopup() {
    popup.classList.remove('active');
    if (overlay) overlay.classList.remove('active');
    document.body.style.overflow = '';
  }

  // Trigger on mouse exit (desktop)
  document.addEventListener('mouseleave', (e) => {
    if (e.clientY <= 5 && !triggered) {
      showTimer = setTimeout(showPopup, 200);
    }
  });

  // Cancel if mouse re-enters
  document.addEventListener('mouseenter', () => {
    clearTimeout(showTimer);
  });

  // Trigger after 45s on mobile (fallback)
  setTimeout(() => {
    if (!triggered) showPopup();
  }, 45000);

  // Close handlers
  if (closeBtn) closeBtn.addEventListener('click', hidePopup);
  if (dismissBtn) dismissBtn.addEventListener('click', hidePopup);
  if (overlay) overlay.addEventListener('click', hidePopup);

  if (ctaBtn) {
    ctaBtn.addEventListener('click', () => {
      hidePopup();
      // Scroll to products
      const target = document.getElementById('eco-home');
      if (target) target.scrollIntoView({ behavior: 'smooth' });
    });
  }

  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape') hidePopup();
  });
}

    btn.addEventListener('click', () => {
      // Navigate to Shopify cart
      window.open(`https://jv17sy-r4.myshopify.com/cart`, '_blank', 'noopener');
    });

}

// ─────────────────────────────────────────────
// NAV ACTIVE LINK ON SCROLL
// ─────────────────────────────────────────────
function initNavHighlight() {
  const sections = ['eco-home', 'eco-pet', 'bundles'];
  const navLinks = document.querySelectorAll('.nav-link');

  const observer = new IntersectionObserver(
    (entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const id = entry.target.id;
          navLinks.forEach(link => {
            link.classList.remove('active');
            if (link.getAttribute('href') === `#${id}`) {
              link.classList.add('active');
            }
          });
        }
      });
    },
    { threshold: 0.3, rootMargin: '-80px 0px 0px 0px' }
  );

  sections.forEach(id => {
    const el = document.getElementById(id);
    if (el) observer.observe(el);
  });

  // Home link active when near top
  window.addEventListener('scroll', debounce(() => {
    if (window.scrollY < 300) {
      navLinks.forEach(link => link.classList.remove('active'));
      const homeLink = document.querySelector('.nav-link[href="#"]');
      if (homeLink) homeLink.classList.add('active');
    }
  }, 100), { passive: true });
}

// ─────────────────────────────────────────────
// LAZY LOAD IMAGES (extra safety)
// ─────────────────────────────────────────────
function initLazyImages() {
  if ('loading' in HTMLImageElement.prototype) return; // native support
  const images = document.querySelectorAll('img[loading="lazy"]');
  const imgObserver = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const img = entry.target;
        img.src = img.dataset.src || img.src;
        imgObserver.unobserve(img);
      }
    });
  });
  images.forEach(img => imgObserver.observe(img));
}

/**
 * Normalize a raw product object to ensure required fields
 */
function normalizeProduct(p) {
  return {
    id: p.id || p.variant_id || String(Math.random()).slice(2),
    title: p.title || p.name || 'Eco Product',
    price: parseFloat(p.price) || 0,
    compare_at_price: p.compare_at_price ? parseFloat(p.compare_at_price) : null,
    image: p.image || p.featured_image || p.images?.[0] || '',
    collection: (p.collection || p.type || p.product_type || '').toLowerCase(),
    rating: parseFloat(p.rating) || 4.5,
    review_count: parseInt(p.review_count || p.reviews_count || 0),
    badge: p.badge || null,
    is_best_seller: Boolean(p.is_best_seller || p.best_seller),
    is_new: Boolean(p.is_new || p.new_arrival),
    tags: Array.isArray(p.tags) ? p.tags : [],
  };
}
